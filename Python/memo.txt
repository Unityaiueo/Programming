【計算】

# + - * / // % **

x = 10
print(x / 3) # 3.33...
print(x // 3) # 3　切り捨て割り
print(x % 3) # 1
print(x ** 2) # 100　２乗

y = 4
# y = y + 12
y += 12
print(y) # 16

# and or not
print(True and False) # False　両方trueならtrue
print(True or False) # True どちらかがtrueならtrue
print(not True) # False trueでないならfalse
